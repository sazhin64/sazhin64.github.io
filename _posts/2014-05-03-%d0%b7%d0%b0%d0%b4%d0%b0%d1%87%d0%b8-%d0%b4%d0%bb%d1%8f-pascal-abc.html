Встретил недавно задачку, в которой требуется нарисовать треугольник по вводимым с клавиатуры координатам. Она легко реализуется с помощью модуля GraphABC и процедуры line.

Программа рисования треугольника,</strong> по вводимым с клавиатуры координатам.

Вариант 1

uses graphabc;

var x,y,x1,y1,x2,y2:integer;

begin

readln(x,y,x1,y1,x2,y2);

line(x,y,x1,y1);

line(x1,y1,x2,y2);

line(x2,y2,x,y);

end.</code>

Вариант 2.

uses graphabc;

var x,y,x1,y1,x2,y2:integer;

begin

readln(x,y,x1,y1,x2,y2);

moveto(x,y);

lineto(x1,y1);

lineto(x2,y2);

lineto(x,y);

end.</code>

Вариант 3.

uses graphabc;

var x,y,x1,y1,x2,y2:integer;

begin

readln(x,y,x1,y1,x2,y2);

moveto(x,y);

lineto(x1,y1);

lineto(x2,y2);

lineto(x,y);

FloodFill(x+5,y+1,clred); {закрашивает треугольник красным, или вокруг него}

end.</code>

Программа запрашивает координаты трех точек и высчитывает длины сторон треугольника, построенного по этим точкам, а также рисует его на координатной плоскости.
uses graphabc;
var x1,y1,x2,y2,x3,y3:integer;
a,b,c:real;
begin
readln(x1,y1,x2,y2,x3,y3);
a:=sqrt(sqr(x1-x2)+sqr(y1-y2));
b:=sqrt(sqr(x2-x3)+sqr(y2-y3));
c:=sqrt(sqr(x3-x1)+sqr(y3-y1));
writeln(a,' ',b,' ',c);
line(0,200,500,200);
line(200,0,200,500);
line(200+x1,200-y1,200+x2,200-y2);
line(200+x2,200-y2,200+x3,200-y3);
line(200+x3,200-y3,200+x1,200-y1);
end.</code>
