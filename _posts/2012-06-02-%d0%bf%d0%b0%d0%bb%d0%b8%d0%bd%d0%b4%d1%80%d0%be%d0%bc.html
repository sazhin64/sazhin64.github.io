Задача: определить, является ли введенное с клавиатуры слово палиндромом?

Решение: сравним первый элемент массива с последним, предпоследний со вторым и т. д.
var a:string;
k,n,i: integer;
begin
writeln ('введите слово');
readln (a);
k:=0;
n:=length(a);
for i:=1 to (n div 2) do
if copy(a,i,1) <> copy(a,n-i+1,1) then k:=1;
if k=0 then writeln ('палиндром')
else writeln ('не палиндром');
end.</code>
