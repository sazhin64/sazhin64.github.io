Очень простой проект. Нужно всего-навсего расположить 1 командную кнопку на форму и написать код. При выполнении, программа рисует оси координат, и относительно осей рисует окружность красного цвета.

Private Sub Command1_Click()
Scale (-10, 7)-(10, -7)
DrawWidth = 2
Line (0, -7)-(0, 7)
Line (-10, 0)-(10, 0)
PSet (9.5, -0.3), QBColor(7)
Print "Y"
PSet (-0.5, 6.9), QBColor(7)
Print "X"
Circle (0, 0), 3, vbRed
End Sub</code>
