Необходимо написать программу, которая выводит на экран все числа Цукермана, то есть трехзначные числа, которые нацело делятся на произведение своих цифр.
Например: 384 3*8*4 = 96 384/96=4
решение:
Вариант 1.
var i,a,b,c:integer;
begin
for i := 111 to 999 do
begin
a := i div 100;
b := (i div 10) mod 10;
c := i mod 10;
if (b <> 0) and (c <> 0) and (i mod (a * b * c) = 0) then writeln(i);
end;
end.</code>

Вариант 2.
var i, a, b, c:integer;
begin
for a := 1 to 9 do
begin
for b := 1 to 9 do
begin
for c := 1 to 9 do
begin
i := a * 100 + b * 10 + c;
if i mod (a * b * c) = 0 then writeln(i);
end;
end;
end;
end.</code>
