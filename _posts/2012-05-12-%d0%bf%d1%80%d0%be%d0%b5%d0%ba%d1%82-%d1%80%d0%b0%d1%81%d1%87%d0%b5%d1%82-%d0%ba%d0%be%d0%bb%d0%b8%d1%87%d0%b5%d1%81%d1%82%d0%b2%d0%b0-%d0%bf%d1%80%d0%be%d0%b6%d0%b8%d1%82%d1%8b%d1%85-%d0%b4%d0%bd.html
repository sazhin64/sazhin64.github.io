В лейбле будет надпись: вы прожили на свете дней в количестве:
Одна кнопка для расчета, другая для выхода.
Чтобы все получилось, необходимо переименовать текстбоксы соответственно: txtdate1, txtdate2, txtresult.
Option Explicit
Dim dtmdate1 As Date
Dim dtmdate2 As Date
Dim dtmdays As Long

Private Sub Command1_Click()
dtmdate1 = CDate(txtdate1.Text)
dtmdate2 = CDate(txtdate2.Text)
dtmdays = dtmdate1 - dtmdate2
txtresult.Text = dtmdays
End Sub
